<script>
const crt=document.getElementById('crt');
const rear=document.getElementById('rear');
const front=document.getElementById('front');
const poly=document.getElementById('poly');
const slider=document.getElementById('move-slider');
let rx=50;const ry=85;
function update(){
rear.style.left=rx+'%';rear.style.top=ry+'%';
const w=crt.clientWidth;const h=crt.clientHeight;
const px=(rx/100)*w;const py=(ry/100)*h;
poly.setAttribute('points',`${px},${py} ${(10/100)*w},0 ${(90/100)*w},0`);
let fx=rx+(50-rx)*(ry-55)/(ry-0);
if(fx<15)fx=15;if(fx>85)fx=85;
front.style.left=fx+'%';front.style.top='55%';
}
slider.addEventListener('input',(e)=>{rx=parseInt(e.target.value);update();});
window.addEventListener('resize',update);
update();
</script>
