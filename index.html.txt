    <div class="info">
        <b>📱 アプリの使い方</b><br>
        ・赤い丸（後衛）を指でドラッグできます。<br>
        ・動かすと、相手コートへの打球範囲が変化します。<br>
        ・青い丸（前衛）は、自動で最適位置へ移動します。<br>
        ・スマホ・タブレット・PC対応です。
    </div>

    <script>

        const crt = document.getElementById('crt');
        const rear = document.getElementById('rear');
        const front = document.getElementById('front');
        const poly = document.getElementById('poly');

        // 後衛初期位置
        let rx = 50;
        let ry = 85;

        // ドラッグ中か
        let dragging = false;

        // 表示更新
        function update() {

            // 後衛位置
            rear.style.left = rx + '%';
            rear.style.top = ry + '%';

            const w = crt.clientWidth;
            const h = crt.clientHeight;

            const px = (rx / 100) * w;
            const py = (ry / 100) * h;

            // 打球エリア（三角形）
            poly.setAttribute(
                'points',
                `
                ${px},${py}
                ${(10/100)*w},0
                ${(90/100)*w},0
                `
            );

            // 前衛位置計算
            let fx = rx + (50 - rx) * 0.55;

            // 制限
            if (fx < 15) fx = 15;
            if (fx > 85) fx = 85;

            front.style.left = fx + '%';
            front.style.top = '55%';
        }

        // プレイヤー移動
        function movePlayer(clientX, clientY) {

            const r = crt.getBoundingClientRect();

            let x =
                ((clientX - r.left) / r.width) * 100;

            let y =
                ((clientY - r.top) / r.height) * 100;

            // 移動範囲制限
            x = Math.max(5, Math.min(95, x));
            y = Math.max(53, Math.min(95, y));

            rx = x;
            ry = y;

            update();
        }

        // ===== タッチ操作 =====

        rear.addEventListener(
            'touchstart',
            (e) => {
                dragging = true;
                e.preventDefault();
            },
            { passive:false }
        );

        window.addEventListener(
            'touchmove',
            (e) => {

                if (!dragging) return;

                movePlayer(
                    e.touches[0].clientX,
                    e.touches[0].clientY
                );

                e.preventDefault();

            },
            { passive:false }
        );

        window.addEventListener(
            'touchend',
            () => {
                dragging = false;
            }
        );

        // ===== マウス操作 =====

        rear.addEventListener(
            'mousedown',
            () => {
                dragging = true;
            }
        );

        window.addEventListener(
            'mousemove',
            (e) => {

                if (!dragging) return;

                movePlayer(
                    e.clientX,
                    e.clientY
                );
            }
        );

        window.addEventListener(
            'mouseup',
            () => {
                dragging = false;
            }
        );

        // リサイズ対応
        window.addEventListener(
            'resize',
            update
        );

        // 初期描画
        update();

    </script>
