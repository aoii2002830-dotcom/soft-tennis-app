                <!-- 相手の打球可能範囲（扇形）を動的に描画 -->

        <div class="info-card">
            <h2>📱 アプリの使い方</h2>
            <ul>
                <li>下の赤い丸<span class="badge badge-red">後衛</span>を指で掴んで動かせます。</li>
                <li>後衛を動かすと、相手コートの「左角」と「右角」へ飛ぶ**打球の広さ（黄色いエリア）**が自動で変化します。</li>
                <li>青い丸<span class="badge badge-blue">前衛</span>は、その打球の広さのちょうど**真ん中（センターライン）の直線上で、ネット際**に自動でポジションを取ります！</li>
                <li>画面下部に余裕を持たせ、操作しやすく調整しました。</li>
            </ul>

    <script>
        const court = document.getElementById('court');
        const rear = document.getElementById('player-rear');
        const front = document.getElementById('player-front');
        const poly = document.getElementById('ball-angle-poly');

        // 初期位置
        let rearX = 50; 
        let rearY = 85; 

        // 相手コートの左右の角（ここに向けて打球が広がると仮定）
        const leftTargetX = 10;  // 左ダブルスライン
        const rightTargetX = 90; // 右ダブルスライン
        const targetY = 0;       // 相手のベースライン

        // 前衛が守る高さ（ネットのすぐ下 Y=55%）
        const frontBaseY = 55;

        function updatePositions() {
            // 後衛の位置反映
            rear.style.left = rearX + '%';
            rear.style.top = rearY + '%';

            // コートの実際のピクセルサイズを取得してSVGの座標を計算
            const w = court.clientWidth;
            const h = court.clientHeight;

            const rx = (rearX / 100) * w;
            const ry = (rearY / 100) * h;
            const ltx = (leftTargetX / 100) * w;
            const rtx = (rightTargetX / 100) * w;
            const ty = (targetY / 100) * h;

            // 写真通りの「打球の広さ（三角形のビーム）」を画面に描く
            poly.setAttribute('points', `${rx},${ry} ${ltx},${ty} ${rtx},${ty}`);

            // 前衛の位置計算（左右のターゲットのちょうど真ん中に向かう直線上に配置）
            const centerTargetX = (leftTargetX + rightTargetX) / 2; // = 50% (センター)
            
            // 後衛からセンターへ向かう直線上の、ネット際(frontBaseY)のX座標を計算
            let calculatedFrontX = rearX + (centerTargetX - rearX) * (rearY - frontBaseY) / (rearY - targetY);

            // 前衛のコート外飛び出し防止
            if (calculatedFrontX < 15) calculatedFrontX = 15;
            if (calculatedFrontX > 85) calculatedFrontX = 85;

            // 前衛の位置反映
            front.style.left = calculatedFrontX + '%';
            front.style.top = frontBaseY + '%';
        }

        // 読み込み時に実行
        window.addEventListener('resize', updatePositions);
        setTimeout(updatePositions, 100);

        // ドラッグ処理
        function handleMove(e) {
            e.preventDefault();
            const rect = court.getBoundingClientRect();
            const clientX = e.touches ? e.touches[0].clientX : e.clientX;
            const clientY = e.touches ? e.touches[0].clientY : e.clientY;

            let x = ((clientX - rect.left) / rect.width) * 100;
            let y = ((clientY - rect.top) / rect.height) * 100;

            // 移動範囲制限（自分のコート側のみ）
            if (x < 5) x = 5;
            if (x > 95) x = 95;
            if (y < 53) y = 53; 
            if (y > 95) y = 95; 

            rearX = x;
            rearY = y;
            updatePositions();
        }

        court.addEventListener('touchstart', (e) => {
            if (e.target === rear) {
                court.addEventListener('touchmove', handleMove, { passive: false });
            }
        });
        court.addEventListener('touchend', () => {
            court.removeEventListener('touchmove', handleMove);
        });
        rear.addEventListener('mousedown', () => {
            court.addEventListener('mousemove', handleMove);
        });
        window.addEventListener('mouseup', () => {
            court.removeEventListener('mousemove', handleMove);
        });
    </script>
