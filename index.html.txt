<script>
const canvas = document.getElementById('courtCanvas');
const ctx = canvas.getContext('2d');
const angleSlider = document.getElementById('angleSlider');
const angleDisplay = document.getElementById('angleDisplay');

// シミュレーションの状態管理
let court = {
    x: 20, y: 30,
    width: 320, height: 400
};

// 相手（奥側）の打点位置（初期値は右奥）
let opponentX = court.x + court.width * 0.7; 
let opponentY = court.y + 10; 

// 自コート（手前側）のベースライン中央（レシーバー・後衛の基準点）
let myPlayerX = court.x + court.width / 2;
let myPlayerY = court.x + court.height + 10;

// 前衛のポジション（自動計算される）
let volleyerX = 0;
let volleyerY = 0;

let isDragging = false;
let spreadAngle = parseInt(angleSlider.value); // 打球の展開角度

// マウス・タッチイベントの登録
canvas.addEventListener('mousedown', startDrag);
canvas.addEventListener('mousemove', drag);
canvas.addEventListener('mouseup', endDrag);
canvas.addEventListener('touchstart', startDrag, {passive: false});
canvas.addEventListener('touchmove', drag, {passive: false});
canvas.addEventListener('touchend', endDrag);

angleSlider.addEventListener('input', (e) => {
    spreadAngle = parseInt(e.target.value);
    angleDisplay.innerText = spreadAngle + '°';
    removePresetActive();
    draw();
});

function startDrag(e) {
    const pos = getMousePos(e);
    // 赤い打点（相手位置）の近くをクリックしたか判定
    const dist = Math.hypot(pos.x - opponentX, pos.y - opponentY);
    if (dist < 20) {
        isDragging = true;
        e.preventDefault();
    }
}

function drag(e) {
    if (!isDragging) return;
    e.preventDefault();
    const pos = getMousePos(e);
    
    // 相手コートの後ろ側の範囲内にドラッグを制限
    opponentX = Math.max(court.x, Math.min(court.x + court.width, pos.x));
    draw();
}

function endDrag() {
    isDragging = false;
}

function getMousePos(e) {
    const rect = canvas.getBoundingClientRect();
    const clientX = e.touches ? e.touches[0].clientX : e.clientX;
    const clientY = e.touches ? e.touches[0].clientY : e.clientY;
    return {
        x: clientX - rect.left,
        y: clientY - rect.top
    };
}

function setPreset(val, btn) {
    angleSlider.value = val;
    spreadAngle = val;
    angleDisplay.innerText = val + '°';
    
    removePresetActive();
    btn.classList.add('active');
    draw();
}

function removePresetActive() {
    document.querySelectorAll('.preset-btn').forEach(b => b.classList.remove('active'));
}

// メインの描画処理
function draw() {
    // 画面クリア
    ctx.clearRect(0, 0, canvas.width, canvas.height);

    // 1. テニスコートの背景とラインの描画
    ctx.fillStyle = '#1e5e3a';
    ctx.fillRect(0, 0, canvas.width, canvas.height);

    ctx.strokeStyle = '#ffffff';
    ctx.lineWidth = 2;

    // 外枠（コート全体）
    ctx.strokeRect(court.x, court.y, court.width, court.height);

    // サービスサイドライン（ダブルス・シングルス兼用としてシンプルに外枠基準で分割）
    let netY = court.y + court.height / 2;
    let serviceLineDist = court.height * 0.28; // サービスラインの位置調整

    // ネット（中央の白線）
    ctx.beginPath();
    ctx.moveTo(court.x - 5, netY);
    ctx.lineTo(court.x + court.width + 5, netY);
    ctx.stroke();

    // サービスライン（手前と奥）
    ctx.beginPath();
    ctx.moveTo(court.x, netY - serviceLineDist);
    ctx.lineTo(court.x + court.width, netY - serviceLineDist);
    ctx.moveTo(court.x, netY + serviceLineDist);
    ctx.lineTo(court.x + court.width, netY + serviceLineDist);
    ctx.stroke();

    // センターライン
    ctx.beginPath();
    ctx.moveTo(court.x + court.width / 2, netY - serviceLineDist);
    ctx.lineTo(court.x + court.width / 2, netY + serviceLineDist);
    ctx.stroke();

    // 前衛の基準ポジションライン（緑の破線）
    ctx.strokeStyle = '#a3e635';
    ctx.setLineDash([4, 4]);
    ctx.beginPath();
    ctx.moveTo(court.x, netY + serviceLineDist * 0.5);
    ctx.lineTo(court.x + court.width, netY + serviceLineDist * 0.5);
    ctx.stroke();
    ctx.setLineDash([]); // 点線リセット

    // コートラベル
    ctx.fillStyle = '#cbd5e1';
    ctx.font = '12px sans-serif';
    ctx.textAlign = 'center';
    ctx.fillText('相手コート（奥）', canvas.width / 2, court.y - 10);
    ctx.fillText('自分コート（手前）', canvas.width / 2, court.y + court.height + 20);
    
    ctx.font = '10px sans-serif';
    ctx.fillText('ネ', court.x + court.width + 12, netY + 4);

    // 2. 計算：打点から自コート両隅への展開角度と「前衛の正しい位置」
    // 打点からセンターライン（セオリーの中心線）
    let angleToCenter = Math.atan2(myPlayerY - opponentY, myPlayerX - opponentX);
    
    // スライダーの角度（ラジアン変換）
    let radSpread = (spreadAngle * Math.PI) / 180;
    let leftAngle = angleToCenter - radSpread / 2;
    let rightAngle = angleToCenter + radSpread / 2;

    // 角度線がベースライン（コート最下部）と交わるX座標を算出
    let length = court.height + 50; 
    let leftX = opponentX + length * Math.cos(leftAngle);
    let leftY = opponentY + length * Math.sin(leftAngle);
    let rightX = opponentX + length * Math.cos(rightAngle);
    let rightY = opponentY + length * Math.sin(rightAngle);

    // 3. 角度範囲（色付きの扇状エリア）の描画
    ctx.fillStyle = 'rgba(239, 68, 68, 0.15)'; // 薄い赤
    ctx.strokeStyle = 'rgba(239, 68, 68, 0.4)';
    ctx.lineWidth = 1.5;
    ctx.beginPath();
    ctx.moveTo(opponentX, opponentY);
    ctx.lineTo(leftX, leftY);
    ctx.lineTo(rightX, rightY);
    ctx.closePath();
    ctx.fill();
    ctx.stroke();

    // 4. セオリーの中心線（青い点線）
    ctx.strokeStyle = '#60a5fa';
    ctx.lineWidth = 2;
    ctx.setLineDash([6, 4]);
    ctx.beginPath();
    ctx.moveTo(opponentX, opponentY);
    ctx.lineTo(myPlayerX, myPlayerY);
    ctx.stroke();
    ctx.setLineDash([]);

    // 5. 前衛（黄色）の位置決定
    // 前衛は「ネットとサービスラインの間」の高さで、青い点線（コースのド真ん中）の上に立つのが基本セオリー
    volleyerY = netY + serviceLineDist * 0.5; 
    // 相似比・直線の方程式から前衛のX座標を算出
    let t = (volleyerY - opponentY) / (myPlayerY - opponentY);
    volleyerX = opponentX + t * (myPlayerX - opponentX);

    // 6. 前衛（黄色）の描画
    ctx.strokeStyle = '#f59e0b';
    ctx.lineWidth = 2;
    ctx.fillStyle = '#fef08a';
    ctx.beginPath();
    ctx.arc(volleyerX, volleyerY, 10, 0, Math.PI * 2);
    ctx.fill();
    ctx.stroke();
    
    // 外
