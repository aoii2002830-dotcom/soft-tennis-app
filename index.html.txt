    <div class="info">
        <b>📱 アプリの使い方</b><br>
        ・赤い丸（後衛）を指で動かせます。<br>
        ・動かすと、相手コートへ飛ぶ打球の広さ（黄色いエリア）が自動で変化します。<br>
        ・青い丸（前衛）は、その打球のど真ん中の直線上のネット際に自動でポジションを取ります。
    </div>
    <script>
        const crt = document.getElementById('crt');
        const rear = document.getElementById('rear');
        const front = document.getElementById('front');
        const poly = document.getElementById('poly');
        let rx = 50, ry = 85;
        function up() {
            rear.style.left = rx + '%'; rear.style.top = ry + '%';
            const w = crt.clientWidth, h = crt.clientHeight;
            const px = (rx / 100) * w, py = (ry / 100) * h;
            poly.setAttribute('points', `${px},${py} ${(10/100)*w},0 ${(90/100)*w},0`);
            let fx = rx + (50 - rx) * (ry - 55) / (ry - 0);
            if (fx < 15) fx = 15; if (fx > 85) fx = 85;
            front.style.left = fx + '%'; front.style.top = '55%';
        }
        window.addEventListener('resize', up); setTimeout(up, 100);
        function move(e) {
            e.preventDefault(); const r = crt.getBoundingClientRect();
            const cx = e.touches ? e.touches[0].clientX : e.clientX;
            const cy = e.touches ? e.touches[0].clientY : e.clientY;
            let x = ((cx - r.left) / r.width) * 100, y = ((cy - r.top) / r.height) * 100;
            if (x < 5) x = 5; if (x > 95) x = 95; if (y < 53) y = 53; if (y > 95) y = 95;
            rx = x; ry = y; up();
        }
        crt.addEventListener('touchstart', (e) => { if (e.target === rear) crt.addEventListener('touchmove', move, { passive: false }); });
        crt.addEventListener('touchend', () => crt.removeEventListener('touchmove', move));
    </script>
