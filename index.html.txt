        const court = document.getElementById('court');
        const rear = document.getElementById('player-rear');
        const front = document.getElementById('player-front');

        // コートサイズに対する初期位置（％）
        let rearX = 50; // 真ん中
        let rearY = 85; // ベースライン付近

        // 相手コートの目標点（右奥の角：X=92%, Y=0%）
        const targetX = 92;
        const targetY = 0;

        // ネットの位置（コート中央 Y=50%）。前衛が守るY座標の基準（少し手前の55%付近）
        const frontBaseY = 56;

        // 位置を画面に反映し、前衛の連動位置を計算する関数
        function updatePositions() {
            // 後衛の表示更新
            rear.style.left = rearX + '%';
            rear.style.top = rearY + '%';

            // 【超重要】連動の計算ロジック
            // 後衛(rearX, rearY) と 相手角(targetX, targetY) を結ぶ直線が、前衛の守る高さ(frontBaseY)と交わるX座標を計算
            // 公式: X = rearX + (targetX - rearX) * (rearY - frontBaseY) / (rearY - targetY)
            // スマホでの見え方をさらに調整するため、ネットに近づくほどボールの角度に狭まるように補正
            
            let calculatedFrontX = rearX + (targetX - rearX) * (rearY - frontBaseY) / (rearY - targetY);

            // 前衛がコート外に飛び出さないようにブレーキ（可動範囲の制限）
            if (calculatedFrontX < 12) calculatedFrontX = 12;
            if (calculatedFrontX > 88) calculatedFrontX = 88;

            // 前衛の表示更新
            front.style.left = calculatedFrontX + '%';
            front.style.top = frontBaseY + '%';
        }

        // 初期配置を実行
        updatePositions();

        // タッチ操作（ドラッグ）のイベント登録
        function handleMove(e) {
            e.preventDefault();
            const rect = court.getBoundingClientRect();
            
            // タッチとマウスの両方に対応

            const clientY = e.touches ? e.touches[0].clientY : e.clientY;

            let y = ((clientY - rect.top) / rect.height) * 100;
